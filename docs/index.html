free search select demo

this component is useful for letting users select something from a large list. in the example search box below, search for a country in Europe.

angular template codez

<input
    type="text"
    class="form-control"
    [formControl]="commit"
    [appTypeahead]="typeahead"
    [displayWith]="displayFn">

  <app-typeahead #typeahead>
    <app-typeahead-option *ngFor="let commit of (filteredOptions | async)" [value]="commit.commit">
      <span class="message">{{commit.commit.message}}</span>
      <span class="sha">{{shortSha(commit.sha)}}</span>
    </app-typeahead-option>
  </app-typeahead>

js codez

  ngOnInit(): void {
    this.commit = new FormControl();

    this.options = this.http.get(`https://api.github.com/repos/storkme/angular-typeahead-stuff/commits`)
      .map((response) => response.json())
      .share();

    this.filteredOptions = Observable.combineLatest(
      this.commit.valueChanges.startWith(null),
      this.options
    )
      .map(([val, options]) => val ? this.filter(val, options) : options.slice());
  }
